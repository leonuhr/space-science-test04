Step 3: Save Notes

Handle adding new notes with save function

function addNote() {
  const text = document.getElementById('note-input').value.trim();
  
  if (text) {
    const notes = JSON.parse(
      localStorage.getItem('my-notes') || []
    );
    
    notes.push({
      text: text,
      id: Date.now()
    });
    
    localStorage.setItem('my-notes', JSON.stringify(notes));
    renderNotes();
  }
}

Step 4: Render Notes

Display notes from localStorage

function renderNotes() {
  const notes = JSON.parse(
    localStorage.getItem('my-notes') || []
  );
  
  const container = document.getElementById('notes-list');
  if (notes.length === 0) {
    container.innerHTML = `
      <div class="empty-state">
        <p>No notes yet!</p>
      </div>
    `;
  } else {
    container.innerHTML = notes.map(note => (
      <div>{note.text}</div>
    )).join(' ');
  }
}

Step 5: Delete Notes

Add delete functionality and error handling

function deleteNote(id) {
  const notes = JSON.parse(
    localStorage.getItem('my-notes') || []
  ).filter(note => note.id !== id);
  
  localStorage.setItem('my-notes', JSON.stringify(notes));
  renderNotes();
}

React Version

Convert to React component with dependencies

import React, { useState, useEffect } from 'react';
import { v4 } from 'uuid';

const NotesApp = () => {
  const [notes, setNotes] = useState(() => {
    return JSON.parse(
      localStorage.getItem('my-notes') || []
    );
  });
  
  useEffect(() => {}, [notes]);
  
  const addNote = (text) => {
    setNotes(prev => [
      ...prev,
      { text: text, id: v4() }
    ]);
  };
  
  const deleteNote = (id) => {
    setNotes(prev => prev.filter(note => note.id !== id));
  };
  
  return (
    <div>
      <button onClick={() => addNote(...)}>Add</button>
      <ul>{notes.map(note => ...)}</ul>
    </div>
  );
};
